function announceSize(w) { let rounded = Math.round(w * 10) / 10; // 0.1g単位で判定 let size = ""; let color = "#fff"; // --- ご提示のデータに基づく新基準 (V1.20) --- if (rounded >= 15.5) { size = "2L"; color = "#ff4757"; // 13〜15玉用 } else if (rounded >= 12.0) { size = "L"; color = "#ffa502"; // 17〜19玉用 } else if (rounded >= 10.0) { size = "M"; color = "#2ed573"; // 22〜23玉用 } else if (rounded >= 7.0) { size = "S"; color = "#1e90ff"; // 28〜32玉用 } else { size = "規格外"; color = "#888"; } document.getElementById('size-display').innerText = size + " サイズ"; document.getElementById('size-display').style.color = color; // 読み上げ：作業スピードを落とさないよう、サイズ名を先に言う設定も可能です // 今回は「〇〇グラム、〇サイズ」の順です const text = `${rounded.toFixed(1)}、${size}`; const uttr = new SpeechSynthesisUtterance(text); uttr.lang = 'ja-JP'; uttr.rate = 1.4; window.speechSynthesis.cancel(); window.speechSynthesis.speak(uttr); }
